// Delete nodes which have greater value on the right side
// 1. Reverse the linkedlist
//2. Commpute save the larges in max and smallest delete karo
// 3. Reverse again the linkedlist

class solution{
  public:
  Node* reverse(Node*root)
  {
      Node*prev=NULL;
      Node* current=root;
      while(current)
      {
          Node* temp=current->next;
          current->next =prev;
          prev=current;
          current=temp;
      }
      return prev;
  }
  
  Node*compute(Node*head)
  {
   head=reverse(head);
   Node*current=head;
   int max=head->data;
   while(head)
   {
       if(head->data>=max)
       {
           max=head->data;
           prev=head;
           head=head->next;
       }
       else
       {
           prev->next = head->next;
           head=prev->next;
       }
   }
   head=reverse(current);
   return head;
  }
  
};
